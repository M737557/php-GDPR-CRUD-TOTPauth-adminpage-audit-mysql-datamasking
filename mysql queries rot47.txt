-- Enable event scheduler
SET GLOBAL event_scheduler = ON;

-- Create weekly maintenance event
DELIMITER $$
CREATE EVENT weekly_maintenance
ON SCHEDULE EVERY 1 WEEK
STARTS CURRENT_TIMESTAMP
DO
BEGIN
    -- Cleanup old records
    DELETE FROM your_table_name 
    WHERE date_created < DATE_SUB(NOW(), INTERVAL 2 YEAR);
    
    -- Update status flags
    UPDATE your_table_name 
    SET relationship_group = 'Archived' 
    WHERE date_created < DATE_SUB(NOW(), INTERVAL 1 YEAR);
END$$
DELIMITER ;

\\ sort based on date_modified

SELECT 
    id, 
    rot47(relation_number) as relation_number, 
    rot47(passport_number) as passport_number, 
    rot47(gender) as gender, 
    rot47(initials) as initials, 
    rot47(first_name) as first_name, 
    rot47(nickname) as nickname, 
    rot47(name_prefix) as name_prefix, 
    rot47(last_name) as last_name, 
    rot47(postal_code) as postal_code, 
    rot47(house_number) as house_number, 
    rot47(street) as street, 
    rot47(city) as city, 
    rot47(language) as language, 
    rot47(relationship_group) as relationship_group, 
    date_created, 
    date_modified 
FROM contacts
ORDER BY date_modified desc;
